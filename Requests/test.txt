            var oHover = document.getElementsByClassName('hoverImg');
            for(var i=0;i<oHover.length;i++){
                oHover[i].onmouseover = function(){
                    this.style.position = "relative";
                    this.parentNode.style.zIndex = "2";
                    this.style.transition = "all .4s";
                    this.style.webkitTransition = "all .4s";
                    this.style.mozTransition = "all .4s";
                    this.style.webkitTransform = "scale(1.2)";
                    this.style.mozTransform = "scale(1.2)";
                    this.style.transform = "scale(1.2)";
                }
                oHover[i].onmouseout = function(){
                    this.parentNode.style.zIndex = "1";
                    this.style.webkitTransform = "scale(1)";
                    this.style.mozTransform = "scale(1)";
                    this.style.transform = "scale(1)";
                }
            }
            var oHoverParent = document.getElementsByClassName('oHoverParent');
            for(var i=0;i<oHoverParent.length;i++){
                oHoverParent[i].onmouseover = function(){
                    this.children[0].style.position = "relative";
                    this.children[0].parentNode.style.zIndex = "2";
                    this.children[0].style.transition = "all .4s";
                    this.children[0].style.webkitTransition = "all .4s";
                    this.children[0].style.mozTransition = "all .4s";
                    this.children[0].style.webkitTransform = "scale(1.2)";
                    this.children[0].style.mozTransform = "scale(1.2)";
                    this.children[0].style.transform = "scale(1.2)";
                }
                oHoverParent[i].onmouseout = function(){
                    this.children[0].parentNode.style.zIndex = "1";
                    this.children[0].style.webkitTransform = "scale(1)";
                    this.children[0].style.mozTransform = "scale(1)";
                    this.children[0].style.transform = "scale(1)";
                }
            }
        </script>
</body>
</html>
